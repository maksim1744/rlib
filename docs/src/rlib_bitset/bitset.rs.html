5#[derive(Clone, Eq, PartialEq)]
6pub struct Bitset<const N: usize> {
7    data: [u64; N],
8}
9
10impl<const N: usize> Bitset<N> {
11    pub fn new() -> Self {
12        Self { data: [0; N] }
13    }
14
15    pub fn from_u64(x: u64) -> Self {
16        let mut data = [0; N];
17        data[0] = x;
18        Self { data }
19    }
20
21    pub fn set(&mut self, x: usize) {
22        self.data[x / 64] |= 1u64 << (x % 64);
23    }
24
25    pub fn remove(&mut self, x: usize) {
26        self.data[x / 64] &= !(1u64 << (x % 64));
27    }
28
29    pub fn flip(&mut self, x: usize) {
30        self.data[x / 64] ^= 1u64 << (x % 64);
31    }
32
33    pub fn test(&self, x: usize) -> bool {
34        ((self.data[x / 64] >> (x % 64)) & 1) > 0
35    }
36
37    pub fn clear(&mut self) {
38        self.data.fill(0);
39    }
40
41    pub fn iter_bits(&self) -> BitsIter<'_, N> {
42        BitsIter::new(&self.data)
43    }
44
45    pub fn count(&self) -> usize {
46        self.data.iter().map(|x| x.count_ones() as usize).sum::<usize>()
47    }
48}

50macro_rules! bin_op {
51    ($trait:ident, $func:ident) => {
52        impl<const N: usize> $trait for &Bitset<N> {
53            type Output = Bitset<N>;
54
55            fn $func(self, rhs: &Bitset<N>) -> Self::Output {
56                let mut result = Bitset::<N>::new();
57                for (i, (x, y)) in self.data.iter().zip(rhs.data.iter()).enumerate() {
58                    result.data[i] = x.$func(y);
59                }
60                result
61            }
62        }
63    };
64}

67bin_op!(BitOr, bitor);
68bin_op!(BitXor, bitxor);

70macro_rules! bin_op_assign {

72        impl<const N: usize> $trait<&Bitset<N>> for Bitset<N> {
73            fn $func(&mut self, rhs: &Bitset<N>) {
74                for (x, y) in self.data.iter_mut().zip(rhs.data.iter()) {
75                    x.$func(y);
76                }
77            }
78        }
79    };
80}
