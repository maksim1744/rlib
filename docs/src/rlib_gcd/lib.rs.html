3pub fn gcd<T: Integer>(a: T, b: T) -> T {
4    let mut a = a.into_abs();
5    let mut b = b.into_abs();
6    while b != T::ZERO {
7        a %= &b;
8        std::mem::swap(&mut a, &mut b);
9    }
10    a
11}

13pub fn lcm<T: Integer>(a: T, b: T) -> T {

18pub fn egcd<T: Integer>(a: T, b: T, c: T) -> Option<(T, T)> {
19    if a == T::ZERO {
20        if c.clone() % &b != T::ZERO {
21            return None;
22        }
23        return Some((T::ZERO, c / &b));
24    }
25    let (y0, x0) = egcd(b.clone() % &a, a.clone(), c)?;
26    Some((x0 - &((b / &a) * &y0), y0))
27}

29pub fn crt<T: Integer + std::ops::Neg<Output = T>>(a1: T, m1: T, a2: T, m2: T) -> Option<T> {

31    let (x, _) = egcd(m1.clone(), -m2.clone(), a2 - &a1)?;
32    let m2 = m2 / &g;
33    let x = (x % &m2 + &m2) % &m2;
34    Some(m1 * &x + &a1)
